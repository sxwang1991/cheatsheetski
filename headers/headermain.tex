\newcommand{\addPdfSection}[1]{
    \setcounter{page}{1}
    \pagestyle{fancy}
    \rhead{\textbf{\textsf{\Huge{#1 \thepage}}}}
    \lfoot{\textsf{\small{\theauthor}}}
    \cfoot{\textsf{\thetitle}}
    \rfoot{\textsf{\textbf{\Huge{#1 \thepage}}}}
    \renewcommand{\headrulewidth}{0.0pt}
    \renewcommand{\footrulewidth}{0.0pt}
    \includepdf[pages={1-},pagecommand={\pagestyle{fancy}},nup=2x1,landscape=false,delta=4 0,scale=0.92]{pdf/#1}
}
