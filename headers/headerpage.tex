\usepackage{listings}

% no indent
\setlength{\parindent}{0cm}

% alternate color table
\rowcolors{1}{gray!30}{white}

% Sans serif for all document
\renewcommand{\familydefault}{\sfdefault}

% titghen the title
\titlespacing*{\subsection}{1pt}{0em}{0em}
\setlength{\textfloatsep}{0pt}

% environment to make a table of command
\NewEnviron{tabcmd}
{%
\keepXColumns
\begin{tabularx}{\linewidth}{l|X}%
\BODY
\end{tabularx}
}

% add line to tabcmd environment
\newcommand{\addline}[2]{%
\texttt{#1} & \textsf{#2} \\ }

% listings package environment to write code
\lstnewenvironment{code}[2]
{
\def\tmpl{#2}\ifx\tmpl\empty
\else
\lstset{language=#2}
\fi
\def\tmp{#1}\ifx\tmp\empty
\else
\subsection*{#1}
\fi
}{}

% add title
\newcommand{\addtitle}[1]{%
\subsection*{#1}
\vspace{-13pt}
}

% listings package configuration
\lstset{aboveskip=0cm, frame=leftline, framesep=0pt}
